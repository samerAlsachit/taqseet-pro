# Cairo Font Download Script
# You need to download the actual Cairo font files manually
# Visit: https://fonts.google.com/specimen/Cairo
# Click "Download family" to get the ZIP file
# Extract and copy these files to assets/fonts/:
# - static/Cairo-Regular.ttf
# - static/Cairo-Bold.ttf  
# - static/Cairo-Light.ttf

# Alternative: Use PowerShell to download
# Add-Type -AssemblyName System.IO.Compression.FileSystem
# [System.IO.Compression.ZipFile]::ExtractToDirectory("Cairo.zip", ".")
